203 else if (MINPRNSBS <= prn && prn <= MAXPRNSBS) sys = E_Sys::SBS;
204 else if (MINPRNQZS <= prn && prn <= MAXPRNQZS) sys = E_Sys::QZS;
205 else return;
206 return;
207 }
208 int found = sscanf(id,"%c%d",&code,&prn_);
209 if (found > 0)

211 switch (code)

213 case 'G': sys = E_Sys::GPS; break;
214 case 'R': sys = E_Sys::GLO; break;
215 case 'E': sys = E_Sys::GAL; break;
216 case 'J': sys = E_Sys::QZS; break;
217 case 'C': sys = E_Sys::BDS; break;
