59// GCC (and maybe clang) can be made to warn about using 0 or NULL when nullptr
60// is available, so Better Enums tries to use nullptr. This passage uses
61// availability of constexpr as a proxy for availability of nullptr, i.e. it
62// assumes that nullptr is available when compiling on the right versions of gcc
63// and clang with the right -std flag. This is actually slightly wrong, because
64// nullptr is also available in Visual C++, but constexpr isn't. This
65// imprecision doesn't matter, however, because VC++ doesn't have the warnings
66// that make using nullptr necessary.
67#ifdef BETTER_ENUMS_HAVE_CONSTEXPR
68# define BETTER_ENUMS_CONSTEXPR_ constexpr
69# define BETTER_ENUMS_NULLPTR nullptr
70#else
71# define BETTER_ENUMS_CONSTEXPR_
72# define BETTER_ENUMS_NULLPTR NULL
73#endif

395BETTER_ENUMS_CONSTEXPR_ static const char *_name_enders = "= \t\n";
396
397BETTER_ENUMS_CONSTEXPR_ inline bool _ends_name(char c, std::size_t index = 0)
398{
399 return
400 c == _name_enders[index] ? true :
401 _name_enders[index] == '\0' ? false :
402 _ends_name(c, index + 1);
403}
404
405BETTER_ENUMS_CONSTEXPR_ inline bool _has_initializer(const char *s,
406 std::size_t index = 0)
407{
408 return
409 s[index] == '\0' ? false :
410 s[index] == '=' ? true :
411 _has_initializer(s, index + 1);
412}
413
414BETTER_ENUMS_CONSTEXPR_ inline std::size_t
415_constant_length(const char *s, std::size_t index = 0)
416{
417 return _ends_name(s[index]) ? index : _constant_length(s, index + 1);
418}
419
420BETTER_ENUMS_CONSTEXPR_ inline char
421_select(const char *from, std::size_t from_length, std::size_t index)
422{
423 return index >= from_length ? '\0' : from[index];
424}
425
426BETTER_ENUMS_CONSTEXPR_ inline char _to_lower_ascii(char c)
427{
428 return c >= 0x41 && c <= 0x5A ? static_cast<char>(c + 0x20) : c;
429}
430
431BETTER_ENUMS_CONSTEXPR_ inline bool _names_match(const char *stringizedName,
432 const char *referenceName,
433 std::size_t index = 0)
434{
435 return
436 _ends_name(stringizedName[index]) ? referenceName[index] == '\0' :
437 referenceName[index] == '\0' ? false :
438 stringizedName[index] != referenceName[index] ? false :
439 _names_match(stringizedName, referenceName, index + 1);
440}
441
442BETTER_ENUMS_CONSTEXPR_ inline bool
443_names_match_nocase(const char *stringizedName, const char *referenceName,
444 std::size_t index = 0)
445{
446 return
447 _ends_name(stringizedName[index]) ? referenceName[index] == '\0' :
448 referenceName[index] == '\0' ? false :
449 _to_lower_ascii(stringizedName[index]) !=
450 _to_lower_ascii(referenceName[index]) ? false :
451 _names_match_nocase(stringizedName, referenceName, index + 1);
452}

456 char *storage, std::size_t count)
457{
458 std::size_t offset = 0;
459
460 for (std::size_t index = 0; index < count; ++index) {
461 trimmed_names[index] = storage + offset;
462
463 std::size_t trimmed_length =
464 std::strcspn(raw_names[index], _name_enders);
465 storage[offset + trimmed_length] = '\0';
466
467 std::size_t raw_length = std::strlen(raw_names[index]);
468 offset += raw_length + 1;
469 }
470}
