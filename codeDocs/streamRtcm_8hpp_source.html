69 static bool LOC(LockTimeInfo& current,
70 LockTimeInfo& previous,
71 pair<uint8_t, uint8_t> cell)
72 {
73 // TODO should we assume continuity or loss of lock when no previous epoch info???
74
75 if (previous. epoch.is_not_a_date_time()) return false;
76 if (current. epoch.is_not_a_date_time()) return true;
77 // From the logic in 3.5.16.3.3 Lock Time Indicator in the RTCM 10403.3 spec
78
79 auto current_lock_time_and_coefficient = current.get_lock_time_and_coefficient(cell);
80
81 auto n = current_lock_time_and_coefficient.first;
82 auto b = current_lock_time_and_coefficient.second;
83
84 auto previous_lock_time_and_coefficient = previous.get_lock_time_and_coefficient(cell);
85
86 auto p = previous_lock_time_and_coefficient.first;
87 auto a = previous_lock_time_and_coefficient.second;
88
89 auto dt = (current.epoch - previous.epoch).total_milliseconds();
90
91 if (p > n) { return true; }
92 else if (p == n && dt >= a) { return true; }
93 else if (p == n && dt < a) { return false; }
94 else if (p < n && b > p && dt >= (n+b-p)) { return true; }
95 // Do we need the possible LOC???
96 else if (p < n && b > p && n < dt && dt < (n+b-p)) { return true; }
97 else if (p < n && b > p && dt <= n) { return false; }
98 else if (p < n && b <= p && dt > n) { return true; }
99 else if (p < n && b <= p && dt <= n) { return false; }
100
101 // Should never get here
102 return true;
103 }
