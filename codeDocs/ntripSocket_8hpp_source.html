32/* Interface to be used for NTRIP version 2, streams for downloading messages.
33* The NtripSteam was ugraded by Alex and now connects and download ansyncronously
34* in the background on a boost service with it's own thread created in the constuctor.
35* The message chunking of NTRIP version 2 has been implemented and receivedDataBuffer
36* is filled with valid RCTM messages in the background without the chunked encoding data.
37* During testing there is some data downloading that is invalid, being not complient with
38* the chunked protocol. This data is not transfered to receivedDataBuffer.
39* As in the previous implementation getData() is called once per epoch of the main loop
40* and the completed and checked RTCM messages are placed in receivedData in a thread safe
41* way. A destructor was added to correctly close socket when the program exits.
42* Addition ansyncronous code was added to connect reconnect and establish a data stream.
43* An example for the boost library documenation was used;

80 "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
81 "abcdefghijklmnopqrstuvwxyz"

83 };
84
85 for (auto n = len / 3; n--; )
86 {
87 out += tab[ (in[0] & 0xfc) >> 2];
88 out += tab[((in[0] & 0x03) << 4) + ((in[1] & 0xf0) >> 4)];
89 out += tab[((in[2] & 0xc0) >> 6) + ((in[1] & 0x0f) << 2)];
90 out += tab[ in[2] & 0x3f];
91 in += 3;
92 }
93
94 switch (len % 3)
95 {
96 case 2:
97 out += tab[ (in[0] & 0xfc) >> 2];
98 out += tab[((in[0] & 0x03) << 4) + ((in[1] & 0xf0) >> 4)];
99 out += tab[ (in[1] & 0x0f) << 2];
100 out += '=';
101 break;
102
103 case 1:
104 out += tab[ (in[0] & 0xfc) >> 2];
105 out += tab[((in[0] & 0x03) << 4)];
106 out += '=';
107 out += '=';
108 break;
109
110 case 0:
111 break;
112 }
113
114 return out;
115 }

139 if (!boost::regex_match(url, matches, re))

155 << "] path=[" << matches[8] << "]";
156
157 URL out;

159 out.url = url;
160
161 string protocol = matches[1];
162
163 out.user = matches[3];
164 out.pass = matches[4];
165 out.host = matches[5];
166 out.port_str = matches[7];
167 out.path = matches[8];
168
169 out.protocol = protocol. empty() ? "http" : protocol;
170 if (out.port_str.empty())
171 {
172 if (out.protocol == "https") out.port_str = "443";
173 else out.port_str = "2101";
174 }
175 out.port = std::stoi(out.port_str);
176
177 return out;
178 }
